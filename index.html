üéØ Quick Start

import { FourJS } from './src/4js.js';

// Create a scene in one line
const scene = new FourJS('#canvas');

// Add a cargo robot
const robot = scene.addCargoRobot();

// Add some animations
scene.animate.rotate(robot, { speed: 0.01 });
scene.animate.float(robot, { amplitude: 0.5 });

// Start the scene
scene.start();
